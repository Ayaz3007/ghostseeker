#include <iostream>

using namespace std;

int main()
{int k, h, c = 0;
cin>>k>>h;
int x[k], y[h];
for(int i = 0; i < k; i++){
    cin>>x[i];
}
for(int j = 0; j < h; j++){
    cin>>y[j];
}
for(int a = 0; a < k; a++){
    for(int b = 0; b < h; b++){
        if(x[a] == y[b]){
            c += 1;
        }
    }
}
cout<<c;
}
