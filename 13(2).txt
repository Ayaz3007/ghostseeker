#include <iostream>
using namespace std;
int f(int x[], int n, int c){
    if(c == n)
        return 0;
    int b[n];
    b[n-1] = x[0];
    for(int i = 1; i < n; i++){
        b[i-1] = x[i];
        cout<<b[i-1]<<" ";
    }
    cout<<b[n-1]<<endl;
    f(b, n, c + 1);
    
}
int main()
{int n;
cin>>n;
int a[n];
for(int i = 0; i < n; i++){
    cin>>a[i];
}
f(a, n, 1);
}
